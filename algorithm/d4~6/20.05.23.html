  1. 7829 보물왕 태혁

    #include <stdio.h>
   
int main(void) {
    int test_case;
    scanf("%d", &test_case);
    for (int t = 1; t <= test_case; t++){
        int n, x;
        int max = -1, min = 100001;
        scanf("%d", &n);
        for (int i = 0; i < n; i++) {
            scanf("%d", &x);
            if (max < x) {
                max = x;
            }
            if (min > x) {
                min = x;
            }
        }
        printf("#%d %d\n", t, min * max);
    }
    return 0;
}
